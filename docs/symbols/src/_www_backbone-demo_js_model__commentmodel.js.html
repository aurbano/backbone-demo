1 /**
  2  * Comment model
  3  * All methods should live here that change the state of a comment
  4  *
  5  * @class CommentModel
  6  * @extends Backbone.Model
  7  * @author Bodnar Istvan <istvan@gawker.com>
  8  */
  9 /*global CommentModel */
 10 var CommentModel = Backbone.Model.extend(
 11 /** @lends CommentModel.prototype */
 12 	{
 13 		/**
 14 		 * Sample method to change the text of a comment model
 15 		 */
 16 		reverseText: function () {
 17 			if (this.has('text') && this.get('text').length > 0) {
 18 				this.set('text', this.get('text').split('').reverse().join(''));
 19 			}
 20 		}
 21 	}
 22 );
